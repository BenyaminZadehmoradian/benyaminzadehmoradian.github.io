%  \cvsection{Title}{rule-fraction}
%  Prints a gray bold section heading followed by a horizontal rule.
\newcommand{\cvsection}[2]{%
  \vspace{0.35cm}%
  \noindent\textcolor{sectiongray}{%
    {\fontsize{11.5pt}{14pt}\selectfont\textbf{#1}}%
    \hspace{0.45em}%
    \textcolor{sectiongray!50}{\rule[0.55ex]{#2\linewidth}{0.45pt}}%
  }%
  \vspace{0.18cm}%
}

%  \cvdegree{Degree}{Date}{Institution}{Location}{Detail label}{Detail value}{GPA}
%  Prints one education entry.
\newcommand{\cvdegree}[7]{%
  {\fontsize{9pt}{13pt}\selectfont
    \textbf{#1}\hfill\textcolor{lightgray}{#2}\\[0.2em]
    \textbf{#2s}% placeholder — overridden below; kept for structure
  }%
}

%  Simpler: \eduentry{Degree}{Date}{School}{City}{ThesisLabel}{ThesisTitle}{Supervisor}{GPA}
\newcommand{\eduentry}[8]{%
  \begingroup
  \fontsize{9pt}{13.5pt}\selectfont
  \textbf{#1}\hfill\textcolor{lightgray}{#2}\\[0.15em]
  \textbf{#3}\hfill\textcolor{lightgray}{#4}\\[0.15em]
  \textbf{#5:} #6\\[0.15em]
  \textbf{Supervisor:} #7\\[0.15em]
  \textbf{GPA:} #8
  \endgroup
}
